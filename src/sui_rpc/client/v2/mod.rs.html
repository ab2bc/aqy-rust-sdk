51    #[allow(clippy::result_large_err)]
52    pub fn new<T>(uri: T) -> Result<Self>
53    where
54        T: TryInto<http::Uri>,
55        T::Error: Into<BoxError>,
56    {
57        let uri = uri
58            .try_into()
59            .map_err(Into::into)
60            .map_err(tonic::Status::from_error)?;
61        let mut endpoint = tonic::transport::Endpoint::from(uri.clone());
62        if uri.scheme() == Some(&http::uri::Scheme::HTTPS) {
63            endpoint = endpoint
64                .tls_config(ClientTlsConfig::new().with_enabled_roots())
65                .map_err(Into::into)
66                .map_err(tonic::Status::from_error)?;
67        }
68        let channel = endpoint
69            .connect_timeout(Duration::from_secs(5))
70            .http2_keep_alive_interval(Duration::from_secs(5))
71            .connect_lazy();
72
73        Ok(Self {
74            uri,
75            channel,
76            auth: Default::default(),
77            max_decoding_message_size: None,
78        })
79    }
