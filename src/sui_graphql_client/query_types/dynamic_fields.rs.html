113impl DynamicFieldValue {
114    /// Returns the JSON representation of the field value, if available.
115    pub fn field_value_json(&self) -> Option<JsonValue> {
116        match self {
117            DynamicFieldValue::MoveObject(mo) => {
118                mo.contents.as_ref().and_then(|mv| mv.json.clone())
119            }
120            DynamicFieldValue::MoveValue(mv) => mv.json.clone(),
121            _ => None,
122        }
123    }
124
125    /// Return the typename and bcs of this dynamic field value.
126    pub fn type_bcs(&self) -> Option<(TypeTag, Vec<u8>)> {
127        match self {
128            DynamicFieldValue::MoveObject(mo) => mo.contents.as_ref().map(|o| {
129                (
130                    TypeTag::from_str(&o.type_.repr.clone()).expect("Invalid TypeTag"),
131                    base64ct::Base64::decode_vec(&o.bcs.0).expect("Invalid Base64"),
132                )
133            }),
134            DynamicFieldValue::MoveValue(mv) => Some((
135                TypeTag::from_str(&mv.type_.repr.clone()).expect("Invalid TypeTag"),
136                base64ct::Base64::decode_vec(&mv.bcs.0).expect("Invalid Base64"),
137            )),
138            _ => None,
139        }
140    }
141}
142
143impl DynamicField {
144    /// Returns the JSON representation of the field value, if available.
145    pub fn field_value_json(&self) -> Option<JsonValue> {
146        self.value.as_ref().and_then(|v| v.field_value_json())
147    }
148}

150impl TryFrom<DynamicField> for DynamicFieldOutput {
151    type Error = error::Error;
152
153    fn try_from(val: DynamicField) -> Result<Self, Self::Error> {
154        let typetag = TypeTag::from_str(
155            val.name
156                .as_ref()
157                .expect("There should be a name in this dynamic field")
158                .type_
159                .repr
160                .as_str(),
161        )?;
162        Ok(DynamicFieldOutput {
163            name: crate::DynamicFieldName {
164                type_: typetag,
165                bcs: base64ct::Base64::decode_vec(val.name.as_ref().unwrap().bcs.0.as_ref())
166                    .unwrap(),
167                json: val.name.as_ref().unwrap().json.clone(),
168            },
169            value_as_json: val.field_value_json(),
170            value: val.value.and_then(|x| x.type_bcs()),
171        })
172    }
173}
